\section{Uvod}

\begin{frame}{Uvod}
\begin{itemize}
    \item Planinarske ture zahtevaju koordinaciju učesnika, vodiča i operatera.
    \item Potrebno je objediniti pretragu, rezervacije i tok ture na jednom mestu.
    \item Razvijen je predlog informacionog sistema za organizaciju tura.
\end{itemize}
\end{frame}

\section{Analiza sistema}

\begin{frame}{Opis procesa}
\begin{enumerate}
    \item Korisnik pretražuje i bira turu.
    \item Vrši rezervaciju i dobija potvrdu.
    \item Vodič realizuje turu i komunicira sa učesnicima.
    \item Nakon ture sledi ocenjivanje i povratne informacije.
\end{enumerate}
\end{frame}

\begin{frame}{Učesnici u sistemu}
\begin{itemize}
    \item Korisnici
    \item Vodiči
    \item Turistički operateri
    \item Administratori
\end{itemize}
\end{frame}
